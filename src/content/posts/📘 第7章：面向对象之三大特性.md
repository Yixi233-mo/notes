### 实战关参考答案
```python
class PromptTemplate:
    """提示词模板基类"""
    
    def __init__(self, template, name=None):
        self._template = template
        self.name = name or "未命名模板"
        self._required_params = self._extract_params(template)
    
    def _extract_params(self, template):
        """提取模板中的参数名（受保护方法）"""
        import re
        return re.findall(r'\{(\w+)\}', template)
    
    def _validate(self, kwargs):
        """验证必填参数（受保护方法）"""
        missing = [p for p in self._required_params if p not in kwargs]
        if missing:
            raise ValueError(f"缺少必填参数: {missing}")
        return True
    
    def render(self, **kwargs):
        """渲染模板（子类可重写）"""
        self._validate(kwargs)
        return self._template.format(**kwargs)
    
    @property
    def template(self):
        """获取模板内容"""
        return self._template
    
    @property
    def required_params(self):
        """获取必填参数列表"""
        return self._required_params.copy()


class SystemPrompt(PromptTemplate):
    """系统提示词模板"""
    
    def __init__(self, template, role="AI助手", name=None):
        super().__init__(template, name)
        self._role = role
        self._description = f"你是一个{role}，请严格遵循以下规则。"
    
    def render(self, **kwargs):
        """重写渲染方法，添加角色前缀"""
        self._validate(kwargs)
        content = self._template.format(**kwargs)
        return f"{self._description}\n\n{content}"
    
    @property
    def role(self):
        return self._role
    
    @role.setter
    def role(self, value):
        self._role = value
        self._description = f"你是一个{value}，请严格遵循以下规则。"


class FewShotPrompt(PromptTemplate):
    """Few-shot提示词模板"""
    
    def __init__(self, template, examples=None, name=None):
        super().__init__(template, name)
        self._examples = examples or []
    
    def render(self, **kwargs):
        """重写渲染方法，添加示例"""
        self._validate(kwargs)
        
        # 构建示例部分
        example_text = ""
        if self._examples:
            example_parts = []
            for ex in self._examples:
                example_parts.append(f"输入：{ex.get('input', '')}")
                example_parts.append(f"输出：{ex.get('output', '')}")
            example_text = "示例：\n" + "\n".join(example_parts) + "\n\n"
        
        content = self._template.format(**kwargs)
        return example_text + content
    
    @property
    def examples(self):
        return self._examples.copy()
    
    def add_example(self, input_text, output_text):
        """添加示例"""
        self._examples.append({"input": input_text, "output": output_text})


# 测试
if __name__ == "__main__":
    # 基类
    base = PromptTemplate("请回答以下问题：{question}")
    print(base.render(question="什么是AI？"))
    
    # 系统提示词
    system = SystemPrompt("请回答：{question}", role="AI编程助手")
    print("\n", system.render(question="什么是Python？"))
    
    # Few-shot
    fewshot = FewShotPrompt("请按格式回答：{question}")
    fewshot.add_example("1+1", "2")
    fewshot.add_example("2+2", "4")
    print("\n", fewshot.render(question="3+3=？"))
```

import time
import random
from datetime import datetime

class BaseLLM:
    """LLM基类"""
    
    def __init__(self, model_name):
        self._model_name = model_name
        self._total_tokens = 0
        self._call_count = 0
    
    def _call_api(self, messages):
        """受保护的API调用方法（子类实现）"""
        raise NotImplementedError
    
    def generate(self, prompt, **kwargs):
        """生成回复（模板方法）"""
        self._call_count += 1
        
        # 构建消息格式
        messages = [{"role": "user", "content": prompt}]
        
        # 调用API
        response = self._call_api(messages, **kwargs)
        
        # 统计token
        tokens = self.count_tokens(response)
        self._total_tokens += tokens
        
        return response
    
    def count_tokens(self, text):
        """估算token数（简单按字数/4）"""
        return len(text) // 4 + 1
    
    @property
    def model_name(self):
        return self._model_name
    
    @property
    def total_tokens(self):
        return self._total_tokens
    
    @property
    def call_count(self):
        return self._call_count


class OpenAILLM(BaseLLM):
    """OpenAI风格LLM"""
    
    def __init__(self, model_name, temperature=0.7, max_tokens=100):
        super().__init__(model_name)
        self._temperature = temperature
        self._max_tokens = max_tokens
        
        # 模拟响应库
        self._responses = {
            "什么是面向对象编程？": "面向对象编程是一种编程范式，使用类和对象来组织代码。",
            "什么是Python？": "Python是一种高级编程语言，以简洁易读著称。",
            "什么是AI？": "AI是人工智能的缩写，指模拟人类智能的计算机系统。"
        }
    
    def _call_api(self, messages, **kwargs):
        """模拟API调用"""
        # 模拟网络延迟
        time.sleep(0.1)
        
        # 获取用户问题
        user_message = messages[-1]["content"]
        
        # 查找响应（模拟）
        response = self._responses.get(user_message)
        if response is None:
            response = f"这是关于'{user_message[:20]}...'的模拟回复。"
        
        # 添加随机变化（模拟温度）
        if random.random() < self._temperature * 0.5:
            response = response + "（略有变化）"
        
        return response
    
    @property
    def temperature(self):
        return self._temperature
    
    @temperature.setter
    def temperature(self, value):
        if 0 <= value <= 1:
            self._temperature = value


class ChatLLM(BaseLLM):
    """对话式LLM，支持多轮对话"""
    
    def __init__(self, model_name):
        super().__init__(model_name)
        self._history = []
        self._system_prompt = "你是一个有用的AI助手。"
    
    def _call_api(self, messages, **kwargs):
        """模拟API调用"""
        time.sleep(0.1)
        
        # 构建完整对话
        full_messages = [{"role": "system", "content": self._system_prompt}] + messages
        
        # 模拟响应
        user_message = messages[-1]["content"]
        
        # 根据历史生成不同回复
        turn_count = len(self._history)
        response = f"（第{turn_count+1}轮对话）这是对'{user_message[:15]}...'的回复。"
        
        # 如果有系统提示，增加前缀
        if "编程" in self._system_prompt and "Python" in user_message:
            response = "作为编程助手，我建议你学习Python的基础语法。"
        
        return response
    
    def add_message(self, role, content):
        """添加消息到历史"""
        self._history.append({"role": role, "content": content})
    
    def generate(self, prompt, **kwargs):
        """重写生成方法，支持多轮对话"""
        # 添加用户消息
        self.add_message("user", prompt)
        
        # 调用API（使用完整历史）
        response = self._call_api(self._history, **kwargs)
        
        # 添加助手回复
        self.add_message("assistant", response)
        
        # 统计token
        tokens = self.count_tokens(response)
        self._total_tokens += tokens
        self._call_count += 1
        
        return response
    
    def clear_history(self):
        """清空对话历史"""
        self._history = []
    
    def get_history(self):
        """获取对话历史"""
        return self._history.copy()
    
    @property
    def system_prompt(self):
        return self._system_prompt
    
    @system_prompt.setter
    def system_prompt(self, value):
        self._system_prompt = value
