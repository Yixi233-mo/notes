### 实战关（40分钟）
**任务**：实现一个模型检查点管理器，用于保存和加载训练过程中的模型权重。

**背景**：AI训练中需要定期保存模型权重（checkpoint），支持断点续训。

**要求**：

1. 实现`CheckpointManager`类：
    - `__init__(self, save_dir)`：指定保存目录
    - `save(self, model_state, optimizer_state, epoch, loss, name=None)`：保存检查点
    - `load(self, name)`：加载检查点
    - `list_checkpoints(self)`：列出所有检查点
    - `get_latest(self)`：获取最新的检查点
2. 保存格式：使用JSON存储元信息（epoch、loss、时间），使用`pickle`或`torch.save`风格的字典存储权重
3. 使用`json.dump()`保存元信息，便于查看
4. 使用`os.makedirs()`确保目录存在

**使用示例**：

```python
# 模拟模型权重
model_state = {"layer1": [1, 2, 3], "layer2": [4, 5, 6]}
optimizer_state = {"lr": 0.001, "momentum": 0.9}

manager = CheckpointManager("./checkpoints")
manager.save(model_state, optimizer_state, epoch=5, loss=0.23)

# 列出检查点
print(manager.list_checkpoints())

# 加载最新的
state = manager.get_latest()
print(f"Epoch: {state['epoch']}, Loss: {state['loss']}")
```

### 实战关参考答案
```python
import os
import json
import time
from datetime import datetime

class CheckpointManager:
    """模型检查点管理器"""
    
    def __init__(self, save_dir="./checkpoints"):
        self.save_dir = save_dir
        os.makedirs(save_dir, exist_ok=True)
    
    def save(self, model_state, optimizer_state, epoch, loss, name=None):
        """保存检查点"""
        if name is None:
            name = f"checkpoint_epoch_{epoch}_{int(time.time())}"
        
        # 构建保存路径
        checkpoint_dir = os.path.join(self.save_dir, name)
        os.makedirs(checkpoint_dir, exist_ok=True)
        
        # 保存元信息（JSON格式，便于查看）
        meta = {
            "epoch": epoch,
            "loss": loss,
            "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            "name": name
        }
        with open(os.path.join(checkpoint_dir, "meta.json"), "w", encoding="utf-8") as f:
            json.dump(meta, f, indent=2, ensure_ascii=False)
        
        # 保存权重（模拟pickle保存）
        weights = {
            "model_state": model_state,
            "optimizer_state": optimizer_state
        }
        # 实际项目中用torch.save或pickle.dump
        with open(os.path.join(checkpoint_dir, "weights.json"), "w", encoding="utf-8") as f:
            json.dump(weights, f, indent=2)
        
        return name
    
    def load(self, name):
        """加载检查点"""
        checkpoint_dir = os.path.join(self.save_dir, name)
        if not os.path.exists(checkpoint_dir):
            raise FileNotFoundError(f"检查点不存在: {checkpoint_dir}")
        
        # 加载元信息
        with open(os.path.join(checkpoint_dir, "meta.json"), "r", encoding="utf-8") as f:
            meta = json.load(f)
        
        # 加载权重
        with open(os.path.join(checkpoint_dir, "weights.json"), "r", encoding="utf-8") as f:
            weights = json.load(f)
        
        return {**meta, **weights}
    
    def list_checkpoints(self):
        """列出所有检查点"""
        checkpoints = []
        for item in os.listdir(self.save_dir):
            item_path = os.path.join(self.save_dir, item)
            if os.path.isdir(item_path):
                meta_path = os.path.join(item_path, "meta.json")
                if os.path.exists(meta_path):
                    with open(meta_path, "r", encoding="utf-8") as f:
                        meta = json.load(f)
                        checkpoints.append(meta)
        # 按时间排序
        return sorted(checkpoints, key=lambda x: x["timestamp"], reverse=True)
    
    def get_latest(self):
        """获取最新的检查点"""
        checkpoints = self.list_checkpoints()
        if not checkpoints:
            return None
        return self.load(checkpoints[0]["name"])


# 测试

    # 模拟保存

    optimizer_state = {"lr": 0.001, "momentum": 0.9}
    
    name = manager.save(model_state, optimizer_state, epoch=5, loss=0.23)
    print(f"保存: {name}")
    
    # 列出检查点
    print("\n所有检查点:")
    for cp in manager.list_checkpoints():
        print(f"  {cp['name']}: epoch={cp['epoch']}, loss={cp['loss']}")
    
    # 加载最新
    latest = manager.get_latest()
    print(f"\n最新检查点: epoch={latest['epoch']}, loss={latest['loss']}")
```

### 挑战关（60分钟）
**任务**：实现一个数据集缓存工具，用于管理大型数据集的加载和缓存。

**背景**：处理大型数据集时，经常需要从磁盘加载数据，缓存机制可以加速重复访问。

**要求**：

1. 实现`DatasetCache`类：
    - `__init__(self, cache_dir="./cache", max_size=5)`：初始化，限制最大缓存文件数
    - `load_data(filepath)`：加载数据，如果已在缓存则直接从缓存读取
    - `preprocess_data(filepath, processor)`：预处理数据并存入缓存
    - `clear_cache()`：清空缓存
    - `get_cache_info()`：返回缓存统计信息
2. 缓存策略：
    - 使用文件名称作为缓存key
    - 缓存超过`max_size`时，删除最久未使用的文件（LRU）
3. 支持分块读写大文件
4. 使用`os.path.getmtime()`判断文件是否更新

**使用示例**：

```python
cache = DatasetCache("./cache", max_size=3)

# 加载数据（模拟）
def load_raw_data(filepath):
    return f"数据内容 from {filepath}"

data = cache.preprocess_data("data1.txt", load_raw_data)
data = cache.preprocess_data("data2.txt", load_raw_data)

print(cache.get_cache_info())
```

### 挑战关参考答案
```python
import os
import json
import time
import shutil
from datetime import datetime

class DatasetCache:
    """数据集缓存管理器（LRU策略）"""
    
    def __init__(self, cache_dir="./cache", max_size=5):
        self.cache_dir = cache_dir
        self.max_size = max_size
        os.makedirs(cache_dir, exist_ok=True)
        self._init_meta()
    
    def _init_meta(self):
        """初始化或加载元数据"""
        meta_path = os.path.join(self.cache_dir, "_meta.json")
        if os.path.exists(meta_path):
            with open(meta_path, "r", encoding="utf-8") as f:
                self.meta = json.load(f)
        else:
            self.meta = {"files": [], "cache_info": {}}
    
    def _save_meta(self):
        """保存元数据"""
        meta_path = os.path.join(self.cache_dir, "_meta.json")
        with open(meta_path, "w", encoding="utf-8") as f:
            json.dump(self.meta, f, indent=2)
    
    def _get_cache_path(self, filepath):
        """获取缓存文件路径"""
        # 使用文件名的hash或简单名称
        base_name = os.path.basename(filepath)
        return os.path.join(self.cache_dir, f"{base_name}.cache")
    
    def _touch(self, filepath):
        """更新访问时间（LRU策略）"""
        cache_path = self._get_cache_path(filepath)
        # 更新访问时间到meta中
        key = os.path.basename(filepath)
        self.meta["cache_info"][key] = {
            "access_time": time.time(),
            "cache_path": cache_path
        }
        # 更新文件列表
        if key not in self.meta["files"]:
            self.meta["files"].append(key)
        
        # 按访问时间排序
        self.meta["files"].sort(
            key=lambda x: self.meta["cache_info"][x]["access_time"],
            reverse=True
        )
        
        # 检查是否超过最大数量
        while len(self.meta["files"]) > self.max_size:
            oldest = self.meta["files"].pop()
            oldest_path = self.meta["cache_info"][oldest]["cache_path"]
            if os.path.exists(oldest_path):
                os.remove(oldest_path)
            del self.meta["cache_info"][oldest]
        
        self._save_meta()
    
    def load_data(self, filepath):
        """加载数据（自动从缓存或原始文件）"""
        key = os.path.basename(filepath)
        cache_path = self._get_cache_path(filepath)
        
        # 检查缓存是否存在
        if os.path.exists(cache_path):
            # 检查源文件是否更新
            if os.path.exists(filepath):
                source_mtime = os.path.getmtime(filepath)
                cache_mtime = os.path.getmtime(cache_path)
                if cache_mtime >= source_mtime:
                    # 缓存有效，更新访问时间
                    self._touch(filepath)
                    with open(cache_path, "r", encoding="utf-8") as f:
                        return f.read()
        
        # 缓存不存在或已过期，从源加载
        if not os.path.exists(filepath):
            raise FileNotFoundError(f"文件不存在: {filepath}")
        
        with open(filepath, "r", encoding="utf-8") as f:
            content = f.read()
        
        # 存入缓存
        with open(cache_path, "w", encoding="utf-8") as f:
            f.write(content)
        
        self._touch(filepath)
        return content
    
    def preprocess_data(self, filepath, processor):
        """预处理并缓存"""
        content = self.load_data(filepath)
        processed = processor(content)
        
        # 缓存预处理结果
        key = os.path.basename(filepath)
        cache_path = os.path.join(self.cache_dir, f"{key}.processed")
        with open(cache_path, "w", encoding="utf-8") as f:
            f.write(processed)
        
        return processed
    
    def clear_cache(self):
        """清空缓存"""
        for item in os.listdir(self.cache_dir):
            if not item.startswith("_"):
                item_path = os.path.join(self.cache_dir, item)
                if os.path.isfile(item_path):
                    os.remove(item_path)
                elif os.path.isdir(item_path):
                    shutil.rmtree(item_path)
        
        self.meta = {"files": [], "cache_info": {}}
        self._save_meta()
    
    def get_cache_info(self):
        """获取缓存统计信息"""
        total_size = 0
        for item in os.listdir(self.cache_dir):
            if not item.startswith("_"):
                item_path = os.path.join(self.cache_dir, item)
                if os.path.isfile(item_path):
                    total_size += os.path.getsize(item_path)
        
        return {
            "cache_dir": self.cache_dir,
            "max_size": self.max_size,
            "cached_files": len(self.meta["files"]),
            "total_size_bytes": total_size,
            "cached_names": self.meta["files"]
        }


# 测试
if __name__ == "__main__":
    # 准备测试文件
    with open("test_data1.txt", "w", encoding="utf-8") as f:
        f.write("这是测试数据1\n包含一些内容")
    with open("test_data2.txt", "w", encoding="utf-8") as f:
        f.write("这是测试数据2\n也是测试内容")
    with open("test_data3.txt", "w", encoding="utf-8") as f:
        f.write("这是测试数据3\n更多内容")

    print(data1)

    print(data2)
    
    print("\n缓存信息:")
    print(cache.get_cache_info())
```

## 四、面试高频题
| 问题 | 参考答案 |
| --- | --- |
| `with open()`的作用和原理？ | 上下文管理器，自动调用`__enter__`和`__exit__`，确保文件关闭 |
| `r`、`w`、`a`模式的区别？ | `r`只读，`w`写入（覆盖），`a`追加（文件末尾写入） |
| 如何处理大文件读取？ | 分块读取：`while chunk := f.read(8192):` |
| 二进制模式和文本模式的区别？ | 二进制`'b'`不进行换行符转换和编码解码，适合图片、权重文件 |

| `os.path.join()`的作用？ | 自动处理路径分隔符（`/`或`\`），跨平台兼容 |
| `os.walk()`的作用？ | 递归遍历目录，返回`(dirpath, dirnames, filenames)` |
| 文件操作可能抛出的异常？ | `FileNotFoundError`、`PermissionError`、`UnicodeDecodeError` |
