### 实战关（40分钟）
**任务**：实现一个深度学习实验管理类，用于管理训练实验的配置、日志和结果。

**背景**：AI训练中需要管理多个实验，每个实验有独立的配置、日志和检查点。

**要求**：

1. 实现`Experiment`类：
    - `__init__(self, name, config=None)`：初始化实验名称和配置
    - `log(self, message)`：记录日志
    - `save_checkpoint(self, epoch, model_state)`：保存检查点
    - `load_checkpoint(self, epoch)`：加载检查点
    - `get_summary(self)`：返回实验摘要
2. 使用类属性存储全局实验列表
3. 使用类方法获取所有实验
4. 使用静态方法格式化时间

**使用示例**：

```python
config = {
    "learning_rate": 0.001,
    "batch_size": 32,
    "epochs": 100
}

exp = Experiment("resnet50_test", config)
exp.log("训练开始")
exp.log("Epoch 1 完成, loss: 0.23")

exp.save_checkpoint(5, {"layer1": [1,2,3]})

print(exp.get_summary())
print(Experiment.list_experiments())
```

### 实战关参考答案
```python
import os
import json
import time
from datetime import datetime

class Experiment:
    """实验管理类"""
    # 类属性：存储所有实验实例
    _experiments = []
    log_dir = "./experiment_logs"
    
    def __init__(self, name, config=None):
        """初始化实验"""
        self.name = name
        self.config = config or {}
        self.logs = []
        self.start_time = time.time()
        self.checkpoints = []
        
        # 记录到全局列表
        Experiment._experiments.append(self)
        
        # 创建日志目录
        os.makedirs(Experiment.log_dir, exist_ok=True)
        
        # 记录初始日志
        self.log(f"实验 {name} 创建")
        self.log(f"配置: {self.config}")
    
    def log(self, message):
        """记录日志"""
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        log_entry = f"[{timestamp}] {message}"
        self.logs.append(log_entry)
        
        # 写入文件
        log_file = os.path.join(Experiment.log_dir, f"{self.name}.log")
        with open(log_file, "a", encoding="utf-8") as f:
            f.write(log_entry + "\n")
        
        print(log_entry)
    
    def save_checkpoint(self, epoch, model_state):
        """保存检查点"""
        checkpoint = {
            "epoch": epoch,
            "model_state": model_state,
            "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
        self.checkpoints.append(checkpoint)
        self.log(f"检查点已保存: epoch={epoch}")
        return checkpoint
    
    def load_checkpoint(self, epoch):
        """加载检查点"""
        for cp in self.checkpoints:
            if cp["epoch"] == epoch:
                self.log(f"加载检查点: epoch={epoch}")
                return cp
        self.log(f"检查点不存在: epoch={epoch}")
        return None
    
    def get_summary(self):
        """获取实验摘要"""
        elapsed = time.time() - self.start_time
        return {
            "name": self.name,
            "config": self.config,
            "log_count": len(self.logs),
            "checkpoint_count": len(self.checkpoints),
            "duration_seconds": round(elapsed, 2),
            "start_time": datetime.fromtimestamp(self.start_time).strftime("%Y-%m-%d %H:%M:%S")
        }
    
    @classmethod
    def list_experiments(cls):
        """列出所有实验"""
        return [exp.name for exp in cls._experiments]
    
    @staticmethod
    def format_time(seconds):
        """格式化时间"""
        if seconds < 60:
            return f"{seconds:.0f}秒"
        elif seconds < 3600:
            return f"{seconds/60:.1f}分钟"
        else:
            return f"{seconds/3600:.1f}小时"


# 测试
if __name__ == "__main__":
    # 创建实验
    config1 = {"lr": 0.001, "batch_size": 32, "epochs": 100}
    exp1 = Experiment("实验A", config1)
    exp1.log("训练开始")
    exp1.log("Epoch 1 完成, loss: 0.23")
    exp1.log("Epoch 2 完成, loss: 0.18")
    exp1.save_checkpoint(2, {"weight": [1, 2, 3]})
    
    config2 = {"lr": 0.01, "batch_size": 64}
    exp2 = Experiment("实验B", config2)
    exp2.log("训练开始")
    
    # 查看摘要
    print("\n=== 实验摘要 ===")
    print(exp1.get_summary())
    
    # 查看所有实验
    print(f"\n所有实验: {Experiment.list_experiments()}")
```

**任务**：实现一个简化版的PyTorch风格的神经网络层基类。

**背景**：理解OOP在深度学习框架中的应用，实现基本的Layer抽象。

**要求**：

2. 实现具体层：
    - `Linear(in_features, out_features)`：全连接层

4. 模拟前向传播计算

**使用示例**：


### 挑战关参考答案
```python
import random

class Module:
    """神经网络层基类"""
    
    def __init__(self):
        self._modules = {}
        self._parameters = {}
    
    def forward(self, x):
        """前向传播（子类重写）"""
        raise NotImplementedError
    
    def __call__(self, x):
        """使实例可调用"""
        return self.forward(x)
    
    def parameters(self):
        """返回所有参数"""
        params = dict(self._parameters)
        for name, module in self._modules.items():
            params.update(module.parameters())
        return params
    
    def add_module(self, name, module):
        """添加子模块"""
        self._modules[name] = module


class Linear(Module):
    """全连接层"""
    
    def __init__(self, in_features, out_features):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        
        # 初始化权重和偏置（随机）
        self._parameters["weight"] = [
            [random.uniform(-0.1, 0.1) for _ in range(in_features)]
            for _ in range(out_features)
        ]
        self._parameters["bias"] = [random.uniform(-0.1, 0.1) for _ in range(out_features)]
    
    def forward(self, x):
        """前向传播：y = Wx + b"""
        if len(x) != self.in_features:
            raise ValueError(f"输入维度错误: 期望{self.in_features}, 实际{len(x)}")
        
        # 矩阵乘法
        weight = self._parameters["weight"]
        bias = self._parameters["bias"]
        
        result = []
        for i in range(self.out_features):
            value = bias[i]
            for j in range(self.in_features):
                value += weight[i][j] * x[j]
            result.append(value)
        
        return result


class ReLU(Module):
    """ReLU激活函数"""
    
    def forward(self, x):
        """ReLU: max(0, x)"""
        if isinstance(x, list):
            return [max(0, val) for val in x]
        return max(0, x)


class Sequential(Module):
    """顺序容器"""
    
    def __init__(self, *modules):
        super().__init__()
        for i, module in enumerate(modules):
            self.add_module(str(i), module)
    
    def forward(self, x):
        """按顺序执行所有模块"""
        for module in self._modules.values():
            x = module(x)
        return x


class MyModel(Module):
    """自定义模型"""
    
    def __init__(self):
        super().__init__()
        self.fc1 = Linear(3, 5)
        self.relu = ReLU()
        self.fc2 = Linear(5, 2)
    
    def forward(self, x):
        x = self.fc1(x)
        x = self.relu(x)
        x = self.fc2(x)
        return x


# 测试
if __name__ == "__main__":
    # 使用Sequential构建模型
    print("=== Sequential方式 ===")
    model = Sequential(
        Linear(3, 5),
        ReLU(),
        Linear(5, 2)
    )
    x = [1.0, 2.0, 3.0]
    output = model(x)
    print(f"输入: {x}")
    print(f"输出: {output}")
    print(f"参数: {model.parameters()}")
    
    # 使用子类方式
    print("\n=== 子类方式 ===")
    model2 = MyModel()
    output2 = model2(x)
    print(f"输入: {x}")
    print(f"输出: {output2}")
```



## 四、面试高频题
| 问题 | 参考答案 |
| --- | --- |
| `self`是什么？ | 代表实例本身，在方法中通过`self`访问实例属性和方法 |
| 类属性和实例属性的区别？ | 类属性所有实例共享，实例属性每个实例独有；类属性通过类名修改对所有实例生效 |
| `@classmethod`和`@staticmethod`的区别？ | 类方法第一个参数是`cls`，可访问类属性；静态方法无特殊参数，仅作工具函数 |
| `__init__`和`__new__`的区别？ | `__new__`负责创建实例，`__init__`负责初始化实例 |
| `__str__`和`__repr__`的区别？ | `__str__`面向用户（可读），`__repr__`面向开发者（可重构） |
| `__slots__`的作用？ | 限制实例可添加的属性，节省内存 |
