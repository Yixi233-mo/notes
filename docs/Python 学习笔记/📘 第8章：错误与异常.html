一、知识点分区与掌握程度要求 ​

🔵 基础概念区 — 要求：熟悉 ​

知识点掌握程度核心要义
异常概念熟悉程序运行时发生的错误，需要捕获和处理
语法错误vs运行时错误熟悉语法错误无法运行，运行时错误可捕获
异常类型熟悉ValueError、TypeError、KeyError、FileNotFoundError等
异常传播熟悉未捕获的异常会逐层向上传递

🟡 核心机制区 — 要求：理解（面试重点） ​

知识点掌握程度核心机制
try-except-finally执行顺序理解try正常执行→finally；try异常→匹配except→finally
else子句时机理解try无异常时执行else
异常继承体系理解所有异常继承自BaseException，Exception是常用基类
raise与自定义异常理解raise主动抛异常，继承Exception自定义异常

🟢 工程实践区 — 要求：精通（AI开发日常必备） ​

知识点掌握程度实践场景
try-except基本捕获精通捕获可能出错的代码块
捕获特定异常精通except ValueError:精确定位问题
多重except精通不同类型的异常不同处理
finally清理资源精通文件关闭、数据库连接释放
获取异常信息精通except Exception as e:获取错误信息
raise主动抛异常精通参数校验失败时抛出
自定义异常精通业务逻辑特定异常

🔴 扩展关联区 — 要求：了解 ​

知识点掌握程度面试可能会问
try-except-else了解知道存在，能看懂即可
异常链raise ... from了解保留原始异常上下文
with与异常了解上下文管理器内部处理异常

任务：为上一章实现的PromptTemplate添加异常处理，使其更加健壮。

背景：AI应用中的提示词模板渲染可能因参数缺失、类型错误而失败，需要友好处理。

要求：

  1. 为PromptTemplate类添加异常处理：
    • render()方法中捕获参数缺失异常
    • 参数类型错误时抛出明确异常
    • 添加日志记录
  2. 实现自定义异常：
    • TemplateError：模板错误基类
    • MissingParameterError：参数缺失异常
    • InvalidParameterError：参数类型错误异常
  3. 实现安全的渲染函数：
    • 返回结果或错误信息
    • 不中断主程序流程
  4. 使用上下文管理器或finally确保资源释放

使用示例：

python
template = PromptTemplate("你好，{name}，你的分数是{score}")

# 正常使用

# 参数缺失
try:
    result = template.render(name="李四")

python
import logging
import re

# ===== 自定义异常 =====
class TemplateError(Exception):
    """模板异常基类"""
    pass

class MissingParameterError(TemplateError):
    """参数缺失异常"""
    def __init__(self, missing_params):
        self.missing_params = missing_params
        super().__init__(f"缺少必填参数: {missing_params}")

class InvalidParameterError(TemplateError):
    """参数类型错误异常"""
    def __init__(self, param_name, expected_type, actual_value):
        self.param_name = param_name
        self.expected_type = expected_type
        self.actual_value = actual_value
        super().__init__(f"参数 {param_name} 类型错误: 期望 {expected_type}，实际 {type(actual_value).__name__}")

# ===== 配置日志 =====
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger("PromptTemplate")

# ===== PromptTemplate =====
class PromptTemplate:
    def __init__(self, template):
        self._template = template
        self._required_params = self._extract_params(template)
    
    def _extract_params(self, template):
        """提取模板中的参数名"""
        return re.findall(r'\{(\w+)\}', template)
    
    def _validate_params(self, kwargs):
        """验证参数"""
        # 检查缺失参数
        missing = [p for p in self._required_params if p not in kwargs]
        if missing:
            raise MissingParameterError(missing)
        
        # 检查参数类型（name和score的特殊处理）
        if "name" in kwargs and not isinstance(kwargs["name"], str):
            raise InvalidParameterError("name", str, kwargs["name"])
        if "score" in kwargs and not isinstance(kwargs["score"], (int, float)):
            raise InvalidParameterError("score", (int, float), kwargs["score"])
        
        return True
    
    def render(self, **kwargs):
        """渲染模板"""
        logger.info(f"开始渲染模板，参数: {kwargs}")
        
        try:
            self._validate_params(kwargs)
            result = self._template.format(**kwargs)
            logger.info(f"渲染成功: {result[:50]}...")
            return result
        except MissingParameterError as e:
            logger.error(f"参数缺失: {e}")
            raise
        except InvalidParameterError as e:
            logger.error(f"参数错误: {e}")
            raise
        except Exception as e:
            logger.error(f"未知错误: {e}")
            raise TemplateError(f"模板渲染失败: {e}")
    
    def safe_render(self, **kwargs):
        """安全渲染，返回(result, error)元组"""
        try:
            result = self.render(**kwargs)
            return result, None
        except TemplateError as e:
            return None, str(e)
        except Exception as e:
            return None, f"未知错误: {e}"
    
    @property
    def required_params(self):
        return self._required_params.copy()


# ===== 测试 =====
if __name__ == "__main__":
    template = PromptTemplate("你好，{name}，你的分数是{score}")
    
    print("=== 正常渲染 ===")
    result, error = template.safe_render(name="张三", score=95)
    print(f"结果: {result}")
    
    print("\n=== 参数缺失 ===")
    result, error = template.safe_render(name="李四")
    if error:
        print(f"错误: {error}")
    
    print("\n=== 参数类型错误 ===")
    result, error = template.safe_render(name=123, score=90)  # name应为字符串
    if error:
        print(f"错误: {error}")
    
    print("\n=== 捕获异常并处理 ===")
    try:
        template.render(name="王五")  # 缺少score
    except MissingParameterError as e:
        print(f"捕获到异常: {e}")

python
import time
import random
import logging
from datetime import datetime

# ===== 日志配置 =====
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger("LLM")

# ===== 异常体系 =====
class LLMError(Exception):
    """LLM异常基类"""
    def __init__(self, message, original_error=None):
        self.message = message
        self.original_error = original_error
        super().__init__(message)

class APIError(LLMError):
    """API调用失败"""
    pass

class RateLimitError(LLMError):
    """速率限制"""
    def __init__(self, retry_after=5):
        self.retry_after = retry_after
        super().__init__(f"请求过于频繁，请在 {retry_after} 秒后重试")

class AuthenticationError(LLMError):
    """认证失败"""
    pass

class TimeoutError(LLMError):
    """超时"""
    pass

class ModelError(LLMError):
    """模型推理错误"""
    pass

# ===== 重试装饰器 =====
def retry(max_attempts=3, base_delay=1, max_delay=10):
    """重试装饰器"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            last_error = None
            delay = base_delay
            
            for attempt in range(1, max_attempts + 1):
                try:
                    logger.info(f"第 {attempt} 次尝试")
                    return func(*args, **kwargs)
                except RateLimitError as e:
                    # 速率限制：等待指定时间
                    wait_time = e.retry_after
                    logger.warning(f"速率限制，等待 {wait_time} 秒")
                    time.sleep(wait_time)
                    last_error = e
                except (APIError, TimeoutError) as e:
                    # 网络/超时错误：指数退避
                    if attempt < max_attempts:
                        wait_time = min(delay * (2 ** (attempt - 1)), max_delay)
                        logger.warning(f"请求失败，{wait_time} 秒后重试: {e}")
                        time.sleep(wait_time)
                        last_error = e
                    else:
                        raise
                except Exception as e:
                    # 其他错误：直接抛出
                    logger.error(f"不可恢复的错误: {e}")
                    raise
            
            # 所有重试失败
            raise last_error
        return wrapper
    return decorator

# ===== RobustLLM =====
class RobustLLM:
    """带异常处理的LLM"""
    
    def __init__(self, model_name, api_key=None):
        self.model_name = model_name
        self.api_key = api_key or "fake_key"
        self._call_count = 0
        self._error_count = 0
        
        # 模拟响应
        self._responses = {
            "什么是Python？": "Python是一种高级编程语言。",
            "什么是AI？": "AI是人工智能。",
            "什么是面向对象编程？": "OOP是一种编程范式。"
        }
    
    def _validate_auth(self):
        """验证认证"""
        if self.api_key == "invalid":
            raise AuthenticationError("API密钥无效")
        return True
    
    def _mock_api_call(self, prompt):
        """模拟API调用"""
        # 模拟随机错误
        rand = random.random()
        
        if rand < 0.05:
            raise RateLimitError(retry_after=2)
        if rand < 0.10:
            raise TimeoutError("请求超时")
        if rand < 0.15:
            raise APIError("内部服务器错误")
        
        # 模拟认证错误
        if self.api_key == "expired":
            raise AuthenticationError("API密钥已过期")
        
        # 模拟模型错误
        if len(prompt) > 100:
            raise ModelError("输入过长")
        
        # 返回响应
        response = self._responses.get(prompt)
        if response is None:
            response = f"这是对 '{prompt[:30]}...' 的模拟回复"
        
        return response
    
    @retry(max_attempts=3, base_delay=1)
    def generate(self, prompt, **kwargs):
        """生成回复（带重试）"""
        logger.info(f"调用 generate，模型: {self.model_name}")
        
        try:
            # 验证认证
            self._validate_auth()
            
            # 调用API
            response = self._mock_api_call(prompt)
            
            # 记录成功
            self._call_count += 1
            logger.info(f"调用成功，第 {self._call_count} 次")
            
            return response
            
        except AuthenticationError as e:
            logger.error(f"认证错误: {e}")
            raise
        except RateLimitError as e:
            logger.warning(f"速率限制: {e}")
            self._error_count += 1
            raise
        except (APIError, TimeoutError) as e:
            logger.warning(f"API错误: {e}")
            self._error_count += 1
            raise
        except ModelError as e:
            logger.error(f"模型错误: {e}")
            self._error_count += 1
            raise
        except Exception as e:
            logger.error(f"未知错误: {e}")
            self._error_count += 1
            raise LLMError(f"LLM调用失败: {e}", original_error=e)
    
    def safe_generate(self, prompt, **kwargs):
        """安全生成，返回(result, error)"""
        try:
            result = self.generate(prompt, **kwargs)
            return result, None
        except RateLimitError as e:
            return None, f"请求过频繁: {e}"
        except AuthenticationError as e:
            return None, f"认证失败: {e}"
        except (APIError, TimeoutError) as e:
            return None, f"API调用失败: {e}"
        except ModelError as e:
            return None, f"模型错误: {e}"
        except LLMError as e:
            return None, f"LLM错误: {e}"
    
    def get_stats(self):
        """获取统计信息"""
        return {
            "model": self.model_name,
            "call_count": self._call_count,
            "error_count": self._error_count,
            "success_rate": round(
                (self._call_count - self._error_count) / self._call_count * 100,
                1
            ) if self._call_count > 0 else 0
        }


# ===== 上下文管理器 =====
class LLMContext:
    """LLM上下文管理器，自动记录调用"""
    
    def __init__(self, llm, operation="inference"):
        self.llm = llm
        self.operation = operation
        self.start_time = None
    
    def __enter__(self):
        self.start_time = time.time()
        logger.info(f"开始 {self.operation}，模型: {self.llm.model_name}")
        return self.llm
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        elapsed = time.time() - self.start_time
        if exc_type is None:
            logger.info(f"{self.operation} 完成，耗时: {elapsed:.2f}秒")
        else:
            logger.error(f"{self.operation} 失败，耗时: {elapsed:.2f}秒，错误: {exc_val}")
        return False  # 不抑制异常


# ===== 测试 =====
if __name__ == "__main__":
    llm = RobustLLM("gpt-3.5")
    
    print("=== 正常调用 ===")
    with LLMContext(llm, "测试调用"):
        result, error = llm.safe_generate("什么是Python？")
        if error:
            print(f"错误: {error}")
        else:
            print(f"结果: {result}")
    
    print("\n=== 模拟重试 ===")
    # 多次调用测试重试
    prompts = ["什么是Python？", "什么是AI？", "什么是面向对象编程？"]
    for p in prompts:
        result, error = llm.safe_generate(p)
        if error:
            print(f"'{p[:10]}...' 失败: {error}")
        else:
            print(f"'{p[:10]}...' 成功: {result[:20]}...")
    
    print(f"\n=== 统计信息 ===")
    print(llm.get_stats())

四、面试高频题 ​

问题参考答案
try-except-else-finally执行顺序？try正常→else→finally；try异常→except→finally
为什么不建议用裸露except:？会捕获所有异常，包括KeyboardInterrupt等不该捕获的，应捕获Exception

自定义异常如何实现？继承Exception，定义__init__
with语句与异常的关系？上下文管理器自动调用__exit__，传入异常信息
什么时候用else子句？执行依赖try成功且不被except捕获的代码
Python异常继承体系？BaseException → Exception → 具体异常
