一、网络基础概念 ​

1.1 网络编程三要素 ​

要素说明作用
IP地址网络中每台计算机的唯一标识找到计算机
端口标识进程的逻辑地址找到计算机中指定的进程（应用软件）
协议通信规则和约定定义数据如何传输

1.2 TCP/IP 协议族 ​

TCP/IP 是一组通信协议的总称，得名于两个核心协议：

  • TCP（传输控制协议）：提供可靠的、面向连接的数据传输
  • IP（网际协议）：负责数据包的路由和寻址

五、UDP 编程 ​

5.1 UDP 特点 ​

特点说明
无连接不建立连接，直接发送数据
不可靠不保证数据到达，不重传丢失数据
高效开销小，适合实时应用
适用场景流媒体、在线游戏、DNS 查询

plain

    │                               │
    │ recvfrom() ──接收数据          │
    │  ←─────────── sendto() ────   │
    │                               │
    │ sendto() ──发送响应            │
    │  ───────────→ recvfrom()      │
    │                               │
    │ close() ──关闭套接字           │

5.3 UDP 代码示例 ​

UDP 服务端：

python
import socket
import threading

# 1. 创建 UDP 套接字
server_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
server_socket.bind(("127.0.0.1", 8080))

print("=" * 50)
print("UDP 服务端已启动")
print("监听地址: 127.0.0.1:8080")
print("输入 'quit' 退出服务端")
print("=" * 50)

# 用来保存客户端地址
client_addr = None

def receive_from_client():
    """持续接收客户端消息"""
    global client_addr
    while True:
        try:
            # 接收数据
            data, addr = server_socket.recvfrom(1024)
            client_addr = addr  # 记录客户端地址
            
            msg = data.decode('utf-8')
            
            # 检查退出
            if msg == 'quit':
                print("\n[客户端请求退出]")
                break
            
            # 显示收到的消息
            print(f"\n[客户端] {msg}")
            print("[服务端] 请输入回复: ", end="", flush=True)
            
        except:
            break

# 启动接收线程
recv_thread = threading.Thread(target=receive_from_client, daemon=True)
recv_thread.start()

# 主线程用于发送消息
while True:
    try:
        # 服务端输入消息
        msg = input("[服务端] 请输入消息: ")
        
        # 退出判断
        if msg.lower() == 'quit':
            if client_addr:
                server_socket.sendto("quit".encode('utf-8'), client_addr)
            break
        
        # 发送消息给客户端
        if client_addr:
            server_socket.sendto(msg.encode('utf-8'), client_addr)
        else:
            print("[提示] 还没有客户端连接")
            
    except KeyboardInterrupt:
        print("\n[服务端被中断]")
        break

# 关闭套接字
server_socket.close()
print("[服务端已关闭]")

UDP 客户端：

python
import socket
import threading

# 1. 创建 UDP 套接字
client_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

server_addr = ("127.0.0.1", 8080)

print("=" * 50)
print("UDP 客户端已启动")
print("连接服务端: 127.0.0.1:8080")
print("输入 'quit' 退出客户端")
print("=" * 50)

def receive_from_server():
    """持续接收服务端消息"""
    while True:
        try:
            # 接收数据（设置超时，避免卡死）
            client_socket.settimeout(1.0)
            data, _ = client_socket.recvfrom(1024)
            msg = data.decode('utf-8')
            
            # 检查退出
            if msg == 'quit':
                print("\n[服务端请求退出]")
                break
            
            # 显示收到的消息
            print(f"\n[服务端] {msg}")
            print("[客户端] 请输入回复: ", end="", flush=True)
            
        except socket.timeout:
            continue
        except:
            break

# 启动接收线程
recv_thread = threading.Thread(target=receive_from_server, daemon=True)
recv_thread.start()

# 主线程用于发送消息
while True:
    try:
        # 客户端输入消息
        msg = input("[客户端] 请输入消息: ")
        
        # 退出判断
        if msg.lower() == 'quit':
            client_socket.sendto("quit".encode('utf-8'), server_addr)
            break
        
        # 发送消息给服务端
        client_socket.sendto(msg.encode('utf-8'), server_addr)
        
    except KeyboardInterrupt:
        print("\n[客户端被中断]")
        break

# 关闭套接字
client_socket.close()
print("[客户端已关闭]")

六、TCP 编程 ​

6.1 TCP 特点 ​

特点说明
面向连接建立连接后才传输数据（三次握手）
可靠传输确认、重传、排序机制保证数据完整
字节流数据作为字节流传输，无边界
适用场景Web 服务、文件传输、邮件

6.2 TCP 三次握手 ​

plain
客户端                       服务端
   │                           │
   │ ─── SYN (seq=x) ────→    │  (1) 请求连接
   │                           │
   │ ←── SYN/ACK (seq=y) ──   │  (2) 确认 + 请求
   │                           │
   │ ─── ACK (seq=x+1) ──→    │  (3) 确认建立
   │                           │
   │ ←──── 数据传输 ──────→   │

6.4 TCP 代码示例 ​

TCP 服务端：

python
import socket

# 创建 TCP 套接字
tcp_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# 绑定 IP 和端口
tcp_socket.bind(("127.0.0.1", 8080))

# 设置监听（最大等待连接数）
tcp_socket.listen(2)
print("TCP 服务端启动，监听端口 8080...")

# 等待客户端连接
client_socket, client_addr = tcp_socket.accept()
print(f"客户端 {client_addr[0]}:{client_addr[1]} 已连接")

while True:
    # 接收数据
    recv_data = client_socket.recv(1024)
    if not recv_data:
        break
    print(f"{client_addr[0]}:{client_addr[1]} >> {recv_data.decode('utf-8')}")
    
    # 发送响应
    client_socket.send("你好".encode("utf-8"))

client_socket.close()
tcp_socket.close()

        print(f"{server_ip}:{server_port} >> {recv_data.decode('utf-8')}")
    except KeyboardInterrupt:
        break

tcp_socket.close()

python
"""
基于 Starlette 构建的 Web API 服务
提供随机名言查询接口
"""

from starlette.applications import Starlette
from starlette.responses import JSONResponse
from starlette.routing import Route
import requests
import uvicorn
import json

# 一言网 API 地址
HITOKOTO_URL = 'https://v1.hitokoto.cn/'


async def get_hitokoto():
    """获取随机名言"""
    try:
        params = {'c': 'a', 'encode': 'json'}
        response = requests.get(HITOKOTO_URL, params=params, timeout=5)
        
        if response.status_code == 200:
            data = response.json()
            return {
                'hitokoto': data['hitokoto'],
                'from': data['from'] or '未知',
                'from_who': data['from_who'] or '未知'
            }
        else:
            return {'error': f'API 请求失败，状态码: {response.status_code}'}
    except requests.RequestException as e:
        return {'error': f'请求错误: {str(e)}'}


async def homepage(request):
    """根路径处理函数"""
    result = await get_hitokoto()
    return JSONResponse(result)


async def health_check(request):
    """健康检查接口"""
    return JSONResponse({'status': 'ok', 'service': 'hitokoto-api'})


# 创建 Starlette 应用
app = Starlette(
    debug=True,
    routes=[
        Route('/', homepage),
        Route('/health', health_check),
    ]
)


if __name__ == "__main__":
    print("启动服务: http://0.0.0.0:8000")
    print("健康检查: http://localhost:8000/health")
    uvicorn.run(app, host='0.0.0.0', port=8000)

十一、易错点总结 ​

易错点正确理解
TCP 比 UDP 快❌ UDP 更快，但不可靠
UDP 可以传输大文件❌ UDP 有大小限制，大文件需分片
客户端端口可以随意指定❌ 需使用动态端口（1024-65535）
recv(1024) 表示接收 1024 个字符❌ 表示接收 1024 字节
TCP 服务端不需要 bind()❌ 服务端必须 bind() 绑定端口
HTTP 请求体中 GET 可以传数据❌ GET 数据在 URL 中，POST 在请求体中

udp.bind(('host', port))
data, addr = udp.recvfrom(1024)
udp.sendto(data, addr)

# 客户端
udp.sendto(data, ('host', port))
data, addr = udp.recvfrom(1024)

# === TCP ===
# 服务端
tcp.bind(('host', port))
tcp.listen(5)
client, addr = tcp.accept()
data = client.recv(1024)
client.send(data)

# 客户端
tcp.connect(('host', port))
tcp.send(data)
data = tcp.recv(1024)

# === HTTP 请求 ===
import requests
response = requests.get(url, params=params)
response = requests.post(url, json=data)
print(response.status_code)
print(response.json())

📌 核心记忆：

  • IP 找计算机，端口 找进程，协议 定规则
  • UDP：无连接、不可靠、高效（适合视频流、游戏）
  • TCP：面向连接、可靠、有序（适合 Web、邮件、文件）
  • Socket 是进程间网络通信的编程接口
  • HTTP 是应用层协议，基于 TCP
  • 状态码：2xx 成功，3xx 重定向，4xx 客户端错误，5xx 服务器错误
