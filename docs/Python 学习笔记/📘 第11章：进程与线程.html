🔵 基础概念区 — 要求：熟悉 ​

知识点掌握程度核心要义
并发与并行熟悉并发是交替执行，并行是同时执行
进程概念熟悉资源分配的基本单位，有独立内存空间
线程概念熟悉任务调度的基本单位，共享进程内存
进程与线程区别熟悉进程独立、开销大；线程共享、开销小

🟡 核心机制区 — 要求：理解（面试重点） ​

知识点掌握程度核心机制
GIL全局解释器锁理解CPython中同一时刻只允许一个线程执行Python字节码，面试必考
互斥锁原理理解保证共享数据操作的原子性，防止数据竞争
进程间通信理解Queue实现进程间数据传递
线程安全问题理解多线程修改共享变量需要加锁

🟢 工程实践区 — 要求：精通（AI开发实际使用场景） ​

知识点掌握程度实践场景
multiprocessing.Process精通数据预处理并行加速

threading.Thread精通API并发请求、IO密集型任务
线程池ThreadPoolExecutor精通管理多个并发任务
互斥锁Lock精通保护共享资源

python
import multiprocessing
import time
import re
from functools import partial

class DataPreprocessor:
    """数据预处理器（多进程）"""
    
    def __init__(self, num_workers=None):
        if num_workers is None:
            num_workers = multiprocessing.cpu_count()
        self.num_workers = num_workers
        self._processed_count = 0
    
    def _worker(self, text, processor):
        """工作进程执行的处理函数"""
        result = processor(text)
        self._processed_count += 1
        return result
    
    def process_batch(self, texts, processor, batch_size=100):
        """批量处理文本（多进程）"""
        if not texts:
            return []
        
        start_time = time.time()
        
        # 分批处理
        batches = [texts[i:i+batch_size] for i in range(0, len(texts), batch_size)]
        
        results = []
        with multiprocessing.Pool(processes=self.num_workers) as pool:
            # 使用map并行处理
            for batch in batches:
                batch_results = pool.map(processor, batch)
                results.extend(batch_results)
        
        elapsed = time.time() - start_time
        print(f"处理完成: {len(texts)} 条文本, 耗时 {elapsed:.2f}秒")
        
        return results
    
    def process_batch_async(self, texts, processor, batch_size=100):
        """异步批量处理（返回AsyncResult）"""
        batches = [texts[i:i+batch_size] for i in range(0, len(texts), batch_size)]
        
        with multiprocessing.Pool(processes=self.num_workers) as pool:
            # 使用map_async异步执行
            async_result = pool.map_async(processor, texts)
            # 可以继续做其他事情...
            results = async_result.get(timeout=60)
        
        return results

# ===== 测试 =====
if __name__ == "__main__":
    def clean_text(text):
        """清洗文本"""
        # 去除非字母字符，转小写
        cleaned = re.sub(r'[^a-zA-Z\s]', '', text)
        return cleaned.lower()
    
    def tokenize(text):
        """简单分词"""
        return text.split()
    
    # 准备测试数据
    texts = [
        "Hello, World! This is a test.",
        "Python 3.9 is great for AI development.",
        "Machine Learning: supervised and unsupervised.",
        "Deep Learning uses neural networks.",
        "Natural Language Processing is fun!"
    ] * 20  # 重复100条
    
    preprocessor = DataPreprocessor(num_workers=4)
    
    print("=== 清洗处理 ===")
    cleaned = preprocessor.process_batch(texts, clean_text)
    print(f"清洗示例: {cleaned[:2]}")
    
    print("\n=== 分词处理 ===")
    tokenized = preprocessor.process_batch(texts, tokenize)
    print(f"分词示例: {tokenized[:2]}")

挑战关（60分钟） ​

任务：实现一个并发API调用器，使用多线程和线程池并发请求。

背景：AI应用需要批量调用API（如OpenAI、HuggingFace），IO密集型任务适合用多线程加速。

要求：

  1. 实现ConcurrentAPICaller类：
    • __init__(self, max_workers=10, retry_count=3)：配置线程池和重试
    • call(self, func, *args, **kwargs)：单次调用
    • call_batch(self, func, items, concurrency=10)：并发批量调用
    • call_stream(self, func, items)：流式调用（生成器）
  2. 实现重试机制（指数退避）
  3. 使用线程池ThreadPoolExecutor
  4. 使用互斥锁保护共享计数器
  5. 处理异常和超时

使用示例：

python

    return f"处理: {text}"

caller = ConcurrentAPICaller(max_workers=5)

挑战关参考答案 ​

python
import time
import random
import threading
from concurrent.futures import ThreadPoolExecutor, as_completed
from collections import deque

class ConcurrentAPICaller:
    """并发API调用器"""
    
    def __init__(self, max_workers=10, retry_count=3, timeout=30):
        self.max_workers = max_workers
        self.retry_count = retry_count
        self.timeout = timeout
        self._success_count = 0
        self._failure_count = 0
        self._lock = threading.Lock()
        self._results = []
    
    def _call_with_retry(self, func, *args, **kwargs):
        """带重试的调用"""
        last_error = None
        for attempt in range(1, self.retry_count + 1):
            try:
                result = func(*args, **kwargs)
                # 成功
                with self._lock:
                    self._success_count += 1
                return result
            except Exception as e:
                last_error = e
                if attempt < self.retry_count:
                    # 指数退避
                    delay = 0.5 * (2 ** (attempt - 1))
                    print(f"重试 {attempt}/{self.retry_count}，等待 {delay:.1f}秒")
                    time.sleep(delay)
        
        # 所有重试失败
        with self._lock:
            self._failure_count += 1
        raise last_error
    
    def call(self, func, *args, **kwargs):
        """单次调用"""
        return self._call_with_retry(func, *args, **kwargs)
    
    def call_batch(self, func, items, concurrency=None, **kwargs):
        """并发批量调用"""
        if concurrency is None:
            concurrency = self.max_workers
        
        results = [None] * len(items)
        self._success_count = 0
        self._failure_count = 0
        
        start_time = time.time()
        
        with ThreadPoolExecutor(max_workers=concurrency) as executor:
            # 提交所有任务
            future_to_idx = {}
            for idx, item in enumerate(items):
                future = executor.submit(self._call_with_retry, func, item, **kwargs)
                future_to_idx[future] = idx
            
            # 收集结果
            for future in as_completed(future_to_idx):
                idx = future_to_idx[future]
                try:
                    results[idx] = future.result(timeout=self.timeout)
                except Exception as e:
                    results[idx] = f"<错误: {str(e)}>"
        
        elapsed = time.time() - start_time
        print(f"批量完成: {len(items)} 项, 成功: {self._success_count}, "
              f"失败: {self._failure_count}, 耗时: {elapsed:.2f}秒")
        
        return results
    
    def call_batch_with_progress(self, func, items, concurrency=None, **kwargs):
        """带进度显示的批量调用"""
        if concurrency is None:
            concurrency = self.max_workers
        
        results = [None] * len(items)
        completed = 0
        total = len(items)
        lock = threading.Lock()
        
        def process_item(idx, item):
            nonlocal completed
            try:
                result = self._call_with_retry(func, item, **kwargs)
                results[idx] = result
            except Exception as e:
                results[idx] = f"<错误: {str(e)}>"
            finally:
                with lock:
                    completed += 1
                    if completed % max(1, total // 20) == 0 or completed == total:
                        print(f"进度: {completed}/{total} ({completed/total*100:.0f}%)")
        
        start_time = time.time()
        
        with ThreadPoolExecutor(max_workers=concurrency) as executor:
            futures = []
            for idx, item in enumerate(items):
                future = executor.submit(process_item, idx, item)
                futures.append(future)
            
            # 等待所有任务完成
            for future in futures:
                future.result(timeout=self.timeout * 2)
        
        elapsed = time.time() - start_time
        print(f"批量完成: {total} 项, 耗时: {elapsed:.2f}秒")
        
        return results
    
    def get_stats(self):
        """获取统计信息"""
        return {
            "success_count": self._success_count,
            "failure_count": self._failure_count,
            "total": self._success_count + self._failure_count,
            "success_rate": round(
                self._success_count / (self._success_count + self._failure_count) * 100,
                1
            ) if (self._success_count + self._failure_count) > 0 else 0
        }

# ===== 测试 =====
if __name__ == "__main__":
    def mock_api(text):
        """模拟API调用"""
        time.sleep(random.uniform(0.1, 0.3))
        if random.random() < 0.03:
            raise Exception("API临时故障")
        return f"处理: {text.upper()}"
    
    caller = ConcurrentAPICaller(max_workers=5, retry_count=2)
    
    print("=== 并发批量调用 ===")
    items = [f"text_{i}" for i in range(20)]
    results = caller.call_batch(mock_api, items, concurrency=5)
    
    for i, result in enumerate(results[:5]):
        print(f"{items[i]} -> {result}")
    
    print(f"\n统计: {caller.get_stats()}")
    
    print("\n=== 带进度批量调用 ===")
    items2 = [f"item_{i}" for i in range(50)]
    results2 = caller.call_batch_with_progress(mock_api, items2, concurrency=10)
    print(f"完成: {len(results2)} 项")

四、面试高频题 ​

问题参考答案
什么是GIL？对Python多线程有什么影响？CPython的全局解释器锁，同一时刻只允许一个线程执行Python字节码。CPU密集型任务多线程无效，IO密集型仍有收益
进程和线程的区别？进程有独立内存空间，开销大；线程共享内存空间，开销小。进程崩溃不影响其他进程，线程崩溃影响整个进程
什么时候用多进程，什么时候用多线程？CPU密集型用多进程（绕过GIL），IO密集型用多线程（GIL在IO时释放）
线程安全问题如何解决？使用互斥锁Lock，确保共享资源操作的原子性
multiprocessing.Queue和Manager().Queue的区别？multiprocessing.Queue用于普通进程通信，Manager().Queue用于进程池中的进程通信
进程池和线程池的区别？进程池管理进程，线程池管理线程；用途类似，底层资源不同
