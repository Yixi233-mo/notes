Python 文件操作学习笔记 ​

一、文件的基本概念 ​

文件是存储在磁盘上的数据集合，可以是文本、图像、音频、视频或程序代码等。

概念说明
文件名包含名称和扩展名（如 test.txt）
扩展名表示文件类型（.txt文本、.jpg图片、.pyPython代码）
文件路径文件在文件系统中的位置
持久化将内存中的数据写入磁盘文件的过程

1.1 文件的分类 ​

类型特点编码示例
纯文本文件有统一编码，可看作长字符串ASCII、UTF-8、GBK 等.txt、.py、.csv
二进制文件无统一编码，由0和1组成无.jpg、.png、.avi、.exe

1.2 文件路径 ​

路径类型说明示例
相对路径从当前位置开始./hello.py（当前目录）、../hello.py（上级目录）
绝对路径从根目录开始E:/Hello/hello.py（Windows）、/home/user/hello.py（Linux/Mac）
python
# 路径表示方式
# Windows 推荐使用正斜杠或原始字符串
path1 = "E:/Hello/hello.py"      # 正斜杠
path2 = r"E:\Hello\hello.py"     # 原始字符串
path3 = "E:\\Hello\\hello.py"    # 转义反斜杠

whence值说明

# 文本模式示例（只能从开头移动）

完整示例：递归遍历并统计

python
import os

def walk_directory(path):
    """递归遍历目录，统计文件和目录数量"""
    file_count = 0
    dir_count = 0
    
    for root, dirs, files in os.walk(path):
        dir_count += len(dirs)
        file_count += len(files)
        print(f"\n目录: {root}")
        print(f"  子目录: {dirs}")
        print(f"  文件: {files}")
    
    print(f"\n总计: {dir_count} 个子目录, {file_count} 个文件")
    return dir_count, file_count

# 遍历当前目录
walk_directory(os.getcwd())

七、文件拷贝案例 ​

7.1 基础版本（一次读取全部） ​

python
def copy_file(source_path, dest_path):
    """拷贝文件（一次性读取全部内容）"""
    with open(source_path, 'rb') as source:
        content = source.read()
    
    with open(dest_path, 'wb') as dest:
        dest.write(content)
    
    print(f"文件已从 {source_path} 拷贝到 {dest_path}")

# 使用示例
copy_file("D:/mv.png", "E:/mv.png")

def copy_file_chunk(source_path, dest_path, chunk_size=1024):
    """拷贝文件（分块读取，内存友好）"""
    with open(source_path, 'rb') as source:
        with open(dest_path, 'wb') as dest:
            while True:
                chunk = source.read(chunk_size)
                if not chunk:  # 读取完毕
                    break
                dest.write(chunk)
    
    print(f"文件已从 {source_path} 拷贝到 {dest_path}")

# 使用示例
copy_file_chunk("D:/mv.png", "E:/mv.png", chunk_size=4096)

7.3 带进度显示版本 ​

python
import os

def copy_file_with_progress(source_path, dest_path, chunk_size=1024*1024):
    """拷贝文件并显示进度"""
    # 获取源文件大小
    total_size = os.path.getsize(source_path)
    copied_size = 0
    
    with open(source_path, 'rb') as source:
        with open(dest_path, 'wb') as dest:
            while True:
                chunk = source.read(chunk_size)
                if not chunk:
                    break
                dest.write(chunk)
                copied_size += len(chunk)
                progress = (copied_size / total_size) * 100
                print(f"\r拷贝进度: {progress:.1f}%", end="")
    
    print(f"\n文件拷贝完成！")
    print(f"源文件: {source_path} ({total_size} 字节)")
    print(f"目标文件: {dest_path}")

# 使用示例
copy_file_with_progress("D:/large_file.zip", "E:/large_file.zip")

8.1 日志文件处理器 ​

python
import os
import time
from datetime import datetime

class LogProcessor:
    """日志文件处理器"""
    
    def __init__(self, log_dir="logs"):
        self.log_dir = log_dir
        self._ensure_log_dir()
    
    def _ensure_log_dir(self):
        """确保日志目录存在"""
        if not os.path.exists(self.log_dir):
            os.makedirs(self.log_dir)
    
    def write_log(self, level, message):
        """写入日志"""
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        log_line = f"[{timestamp}] [{level}] {message}\n"
        
        # 按日期分割日志文件
        log_file = os.path.join(self.log_dir, f"{datetime.now().strftime('%Y%m%d')}.log")
        
        with open(log_file, "a", encoding="utf-8") as f:
            f.write(log_line)
    
    def read_logs(self, date=None, level=None):
        """读取日志"""
        if date is None:
            date = datetime.now().strftime("%Y%m%d")
        
        log_file = os.path.join(self.log_dir, f"{date}.log")
        
        if not os.path.exists(log_file):
            print(f"日志文件不存在: {log_file}")
            return []
        
        logs = []
        with open(log_file, "r", encoding="utf-8") as f:
            for line in f:
                if level is None or f"[{level}]" in line:
                    logs.append(line.rstrip())
        
        return logs
    
    def clean_old_logs(self, days=30):
        """清理旧日志"""
        now = time.time()
        deleted_count = 0
        
        for filename in os.listdir(self.log_dir):
            filepath = os.path.join(self.log_dir, filename)
            if os.path.isfile(filepath):
                # 获取文件修改时间
                mtime = os.path.getmtime(filepath)
                if (now - mtime) > (days * 24 * 3600):
                    os.remove(filepath)
                    deleted_count += 1
                    print(f"已删除旧日志: {filename}")
        
        print(f"共删除 {deleted_count} 个旧日志文件")

# 使用示例
log_processor = LogProcessor()

# 写入日志
log_processor.write_log("INFO", "程序启动")
log_processor.write_log("ERROR", "发生错误：连接超时")
log_processor.write_log("INFO", "程序关闭")

# 读取日志
print("=== 今日日志 ===")
logs = log_processor.read_logs()
for log in logs:
    print(log)

# 读取 ERROR 级别日志
print("\n=== ERROR 日志 ===")
error_logs = log_processor.read_logs(level="ERROR")
for log in error_logs:
    print(log)

# 清理30天前的日志
# log_processor.clean_old_logs(30)

8.2 配置文件管理器 ​

python
import json
import yaml  # 需要安装: pip install pyyaml

class ConfigManager:
    """配置文件管理器"""
    
    def __init__(self, config_file="config.json"):
        self.config_file = config_file
        self.config = self.load()
    
    def load(self):
        """加载配置文件"""
        if not os.path.exists(self.config_file):
            print(f"配置文件不存在，创建默认配置")
            return self._get_default_config()
        
        ext = os.path.splitext(self.config_file)[1].lower()
        
        with open(self.config_file, "r", encoding="utf-8") as f:
            if ext == ".json":
                return json.load(f)
            elif ext in [".yaml", ".yml"]:
                return yaml.safe_load(f)
            else:
                # 简单的 key=value 格式
                config = {}
                for line in f:
                    line = line.strip()
                    if line and not line.startswith("#"):
                        key, value = line.split("=", 1)
                        config[key.strip()] = value.strip()
                return config
    
    def _get_default_config(self):
        """获取默认配置"""
        return {
            "app_name": "MyApp",
            "version": "1.0.0",
            "debug": True,
            "host": "localhost",
            "port": 8080,
            "database": {
                "host": "localhost",
                "port": 3306,
                "user": "root",
                "password": ""
            }
        }
    
    def save(self):
        """保存配置文件"""
        ext = os.path.splitext(self.config_file)[1].lower()
        
        with open(self.config_file, "w", encoding="utf-8") as f:
            if ext == ".json":
                json.dump(self.config, f, indent=4, ensure_ascii=False)
            elif ext in [".yaml", ".yml"]:
                yaml.dump(self.config, f, allow_unicode=True)
            else:
                for key, value in self.config.items():
                    if not isinstance(value, dict):
                        f.write(f"{key}={value}\n")
        
        print(f"配置已保存到: {self.config_file}")
    
    def get(self, key, default=None):
        """获取配置值（支持点号分隔的嵌套键）"""
        keys = key.split(".")
        value = self.config
        
        for k in keys:
            if isinstance(value, dict) and k in value:
                value = value[k]
            else:
                return default
        
        return value
    
    def set(self, key, value):
        """设置配置值（支持点号分隔的嵌套键）"""
        keys = key.split(".")
        target = self.config
        
        for k in keys[:-1]:
            if k not in target:
                target[k] = {}
            target = target[k]
        
        target[keys[-1]] = value
        self.save()

# 使用示例
config = ConfigManager("app_config.json")

print(f"应用名称: {config.get('app_name')}")
print(f"数据库主机: {config.get('database.host')}")

# 修改配置
config.set("debug", False)
config.set("database.password", "secret123")

    content = f.read()  # 忽略无法解码的字符

with open("test.txt", "r", encoding="utf-8", errors="replace") as f:
    content = f.read()  # 用 � 替换无法解码的字符

import chardet

    raw_data = f.read()
    result = chardet.detect(raw_data)
    encoding = result['encoding']
    print(f"检测到的编码: {encoding}")

十、易错点总结 ​

易错点正确理解
忘记关闭文件使用 with 语句自动管理
文本模式读取二进制文件用 'rb' 模式读取二进制文件
编码不一致导致乱码读写时使用相同的编码（推荐 UTF-8）
使用相对路径不确定使用 os.path.abspath() 获取绝对路径
w 模式会清空文件需要追加内容用 'a' 模式
文件不存在时 'r' 模式报错使用 'w' 或 'a' 会自动创建
read() 大文件内存溢出分块读取或逐行读取
路径拼接使用字符串加法使用 os.path.join()
python
# 错误示例
# 1. 忘记关闭文件
f = open("test.txt", "w")
f.write("hello")
# 没有 close()，数据可能未写入磁盘

# 2. 编码问题
# with open("test.txt", "r") as f:  # 默认使用系统编码，可能出错
#     content = f.read()

# 正确做法
with open("test.txt", "r", encoding="utf-8") as f:
    content = f.read()

# 3. 路径拼接错误
path = "E:/folder/" + "subfolder/" + "file.txt"  # 可能多或少斜杠

# 正确做法
import os
path = os.path.join("E:", "folder", "subfolder", "file.txt")
