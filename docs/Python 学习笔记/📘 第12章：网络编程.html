任务：实现一个LLM API调用工具，封装常见的API请求逻辑。

python
import requests
import json
import time

class LLMClient:
    """LLM API调用客户端"""
    
    def __init__(self, base_url, api_key, timeout=30):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.timeout = timeout
        self._headers = {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {api_key}"
        }
        self._call_count = 0
    
    def _request(self, endpoint, method="GET", data=None, stream=False):
        """底层请求方法"""
        url = f"{self.base_url}/{endpoint.lstrip('/')}"
        self._call_count += 1
        
        try:
            if method.upper() == "GET":
                response = requests.get(
                    url,
                    headers=self._headers,
                    timeout=self.timeout,
                    stream=stream
                )
            elif method.upper() == "POST":
                response = requests.post(
                    url,
                    headers=self._headers,
                    json=data,
                    timeout=self.timeout,
                    stream=stream
                )
            else:
                raise ValueError(f"不支持的方法: {method}")
            
            response.raise_for_status()
            return response
            
        except requests.exceptions.Timeout:
            raise Exception(f"请求超时: {url}")
        except requests.exceptions.ConnectionError:
            raise Exception(f"连接失败: {url}")
        except requests.exceptions.HTTPError as e:
            status = response.status_code
            if status == 401:
                raise Exception("API密钥无效或已过期")
            elif status == 429:
                raise Exception("请求过于频繁，请稍后重试")
            elif status == 500:
                raise Exception("服务器内部错误")
            else:
                raise Exception(f"HTTP错误 {status}: {response.text}")
        except Exception as e:
            raise Exception(f"请求失败: {e}")
    
    def chat(self, messages, model="gpt-3.5-turbo", **kwargs):
        """对话请求"""
        data = {
            "model": model,
            "messages": messages,
            **kwargs
        }
        response = self._request("chat/completions", "POST", data)
        result = response.json()
        return result.get("choices", [{}])[0].get("message", {}).get("content")
    
    def embed(self, texts, model="text-embedding-ada-002"):
        """向量化请求"""
        if isinstance(texts, str):
            texts = [texts]
        data = {
            "model": model,
            "input": texts
        }
        response = self._request("embeddings", "POST", data)
        result = response.json()
        return [item.get("embedding") for item in result.get("data", [])]
    
    def stream_chat(self, messages, model="gpt-3.5-turbo", **kwargs):
        """流式对话（生成器）"""
        data = {
            "model": model,
            "messages": messages,
            "stream": True,
            **kwargs
        }
        response = self._request("chat/completions", "POST", data, stream=True)
        
        for line in response.iter_lines():
            if line:
                line = line.decode("utf-8")
                if line.startswith("data: "):
                    if line == "data: [DONE]":
                        break
                    try:
                        chunk = json.loads(line[6:])
                        content = chunk.get("choices", [{}])[0].get("delta", {}).get("content")
                        if content:
                            yield content
                    except json.JSONDecodeError:
                        continue
    
    def get_stats(self):
        """获取统计信息"""
        return {
            "call_count": self._call_count
        }

python
import json
import time
import random
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed

class ModelClient:
    """模型服务客户端"""
    
    def __init__(self, config_path):
        with open(config_path, "r", encoding="utf-8") as f:
            config = json.load(f)
        
        self.service_urls = config.get("service_urls", [])
        self.api_key = config.get("api_key", "")
        self.timeout = config.get("timeout", 30)
        self.retry_count = config.get("retry_count", 3)
        self.models = config.get("models", {})
        self._current_idx = 0
        self._call_log = []
    
    def _get_next_url(self):
        """负载均衡：轮询"""
        url = self.service_urls[self._current_idx % len(self.service_urls)]
        self._current_idx += 1
        return url
    
    def _request(self, endpoint, data, retry_count=None):
        """发送请求（带重试）"""
        if retry_count is None:
            retry_count = self.retry_count
        
        url = self._get_next_url() + endpoint
        headers = {
            "Content-Type": "application/json",
            "Authorization": f"Bearer {self.api_key}"
        }
        
        last_error = None
        for attempt in range(1, retry_count + 1):
            try:
                response = requests.post(
                    url,
                    headers=headers,
                    json=data,
                    timeout=self.timeout
                )
                response.raise_for_status()
                
                # 记录日志
                self._call_log.append({
                    "time": time.time(),
                    "url": url,
                    "status": "success",
                    "attempt": attempt
                })
                
                return response.json()
                
            except requests.exceptions.Timeout:
                last_error = "超时"
                delay = 0.5 * (2 ** (attempt - 1))
                time.sleep(delay)
            except requests.exceptions.ConnectionError:
                last_error = "连接失败"
                delay = 0.5 * (2 ** (attempt - 1))
                time.sleep(delay)
            except requests.exceptions.HTTPError as e:
                last_error = f"HTTP错误: {e}"
                if response.status_code in [401, 403]:
                    break  # 认证错误不重试
                delay = 0.5 * (2 ** (attempt - 1))
                time.sleep(delay)
            except Exception as e:
                last_error = str(e)
                break
        
        self._call_log.append({
            "time": time.time(),
            "url": url,
            "status": "failed",
            "error": last_error
        })
        
        raise Exception(f"请求失败 (重试{retry_count}次): {last_error}")
    
    def predict(self, inputs, model_name=None, **kwargs):
        """同步预测"""
        if model_name is None:
            model_name = list(self.models.keys())[0]
        
        model_config = self.models.get(model_name)
        if model_config is None:
            raise ValueError(f"模型 '{model_name}' 未配置")
        
        endpoint = model_config.get("endpoint", f"/predict/{model_name}")
        data = {"inputs": inputs, **kwargs}
        
        result = self._request(endpoint, data)
        return result.get("result", result)
    
    def batch_predict(self, inputs_list, model_name=None, max_workers=5, **kwargs):
        """批量预测（多线程）"""
        results = [None] * len(inputs_list)
        
        def process(idx, inputs):
            try:
                result = self.predict(inputs, model_name, **kwargs)
                return idx, result, None
            except Exception as e:
                return idx, None, str(e)
        
        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = [
                executor.submit(process, idx, inputs)
                for idx, inputs in enumerate(inputs_list)
            ]
            
            for future in as_completed(futures):
                idx, result, error = future.result()
                if error:
                    results[idx] = f"<错误: {error}>"
                else:
                    results[idx] = result
        
        return results
    
    def get_model_info(self, model_name):
        """获取模型信息"""
        model_config = self.models.get(model_name)
        if model_config is None:
            return None
        return {
            "name": model_name,
            "endpoint": model_config.get("endpoint"),
            "config": model_config.get("config", {})
        }
    
    def get_stats(self):
        """获取统计信息"""
        total = len(self._call_log)
        success = sum(1 for log in self._call_log if log["status"] == "success")
        return {
            "total_requests": total,
            "success_requests": success,
            "failure_requests": total - success,
            "success_rate": round(success / total * 100, 1) if total > 0 else 0
        }


# ===== 测试 =====
def test_model_client():
    # 创建模拟配置
    config = {
        "service_urls": ["http://localhost:8000", "http://localhost:8001"],
        "api_key": "test-key",
        "timeout": 5,
        "retry_count": 2,
        "models": {
            "bert": {"endpoint": "/predict/bert", "config": {"max_length": 512}},
            "gpt": {"endpoint": "/predict/gpt", "config": {"max_tokens": 100}}
        }
    }
    
    with open("test_config.json", "w") as f:
        json.dump(config, f)
    
    # 模拟请求的客户端
    class MockModelClient(ModelClient):
        def _request(self, endpoint, data, retry_count=None):
            # 模拟：90%成功
            if random.random() < 0.1:
                raise Exception("模拟服务故障")
            return {"result": f"处理: {data.get('inputs')}"}
    
    client = MockModelClient("test_config.json")
    
    print("=== 单次预测 ===")
    result = client.predict("Hello world", model_name="bert")
    print(f"结果: {result}")
    
    print("\n=== 批量预测 ===")
    inputs = [f"text_{i}" for i in range(10)]
    results = client.batch_predict(inputs, model_name="gpt", max_workers=3)
    for i, r in enumerate(results[:5]):
        print(f"  {inputs[i]} -> {r}")
    
    print(f"\n=== 统计信息 ===")
    print(client.get_stats())
    
    print("\n=== 模型信息 ===")
    print(client.get_model_info("bert"))


if __name__ == "__main__":
    test_model_client()

四、面试高频题 ​

问题参考答案
requests.get()和requests.post()的区别？GET用于获取数据，参数在URL中；POST用于提交数据，数据在请求体中
什么是HTTP状态码200、400、500？200成功，400客户端错误，500服务器错误
如何设置请求超时？requests.get(url, timeout=5)
如何处理响应JSON？response.json()
什么是RESTful API？用HTTP方法（GET/POST/PUT/DELETE）操作资源的设计风格
调用API时如何处理认证？在Header中添加Authorization: Bearer {token}
Socket和HTTP的关系？Socket是底层通信，HTTP是应用层协议，基于TCP Socket
