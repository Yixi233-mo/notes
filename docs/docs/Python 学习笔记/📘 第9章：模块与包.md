def tokenize(text):
    """简单分词"""
    return text.lower().split()

def pad_sequence(seq, max_len, pad_value=0):
    """填充序列到固定长度"""
    if len(seq) >= max_len:
        return seq[:max_len]
    return seq + [pad_value] * (max_len - len(seq))

def clean_text(text):
    """清洗文本"""

from .preprocessing import tokenize, pad_sequence

class Dataset:
    """数据集类"""
    def __init__(self, data, labels, max_len=128):
        self.data = data
        self.labels = labels
        self.max_len = max_len
        self.size = len(data)
    
    def __len__(self):
        return self.size
    
    def __getitem__(self, idx):
        """获取单个样本"""
        text = self.data[idx]
        tokens = tokenize(text)
        tokens = pad_sequence(tokens, self.max_len)
        return tokens, self.labels[idx]
    
    def get_batch(self, batch_size):
        """获取一个batch"""
        import random
        indices = random.sample(range(self.size), min(batch_size, self.size))
        batch_x = []
        batch_y = []
        for idx in indices:
            x, y = self[idx]
            batch_x.append(x)
            batch_y.append(y)
        return batch_x, batch_y

def load_dataset(path):
    """加载数据集"""
    import json
    with open(path, "r", encoding="utf-8") as f:
        data = json.load(f)
    return Dataset(data["texts"], data["labels"])

# ===== utils/metrics.py =====
import math

def accuracy(y_true, y_pred):
    """准确率"""
    correct = sum(1 for a, b in zip(y_true, y_pred) if a == b)
    return correct / len(y_true)

def precision(y_true, y_pred, positive_label=1):
    """精确率"""
    tp = sum(1 for a, b in zip(y_true, y_pred) if a == b == positive_label)
    fp = sum(1 for a, b in zip(y_true, y_pred) if a != positive_label and b == positive_label)
    if tp + fp == 0:
        return 0.0
    return tp / (tp + fp)

def recall(y_true, y_pred, positive_label=1):
    """召回率"""
    tp = sum(1 for a, b in zip(y_true, y_pred) if a == b == positive_label)
    fn = sum(1 for a, b in zip(y_true, y_pred) if a == positive_label and b != positive_label)
    if tp + fn == 0:
        return 0.0
    return tp / (tp + fn)

def f1_score(y_true, y_pred, positive_label=1):
    """F1分数"""
    p = precision(y_true, y_pred, positive_label)
    r = recall(y_true, y_pred, positive_label)
    if p + r == 0:
        return 0.0
    return 2 * p * r / (p + r)

if __name__ == "__main__":
    y_true = [1, 0, 1, 1, 0, 1]
    y_pred = [1, 0, 0, 1, 0, 1]
    print(f"Accuracy: {accuracy(y_true, y_pred):.2f}")
    print(f"Precision: {precision(y_true, y_pred):.2f}")
    print(f"Recall: {recall(y_true, y_pred):.2f}")
    print(f"F1: {f1_score(y_true, y_pred):.2f}")

# ===== model_plugin/base.py =====
class BaseModel:
    """模型基类"""
    def __init__(self, config=None):
        self.config = config or {}
        self._name = self.__class__.__name__
    
    def forward(self, x):
        raise NotImplementedError
    
    def predict(self, x):
        return self.forward(x)
    
    def get_name(self):
        return self._name


# ===== model_plugin/registry.py =====
import os
import importlib
import inspect

class ModelRegistry:
    """模型注册表"""
    _instance = None
    _models = {}
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._models = {}
        return cls._instance
    
    def register(self, name):
        """注册装饰器"""
        def decorator(cls):
            if not inspect.isclass(cls):
                raise TypeError(f"需要类，获得 {type(cls)}")
            if not issubclass(cls, BaseModel):
                raise TypeError(f"{cls.__name__} 不是 BaseModel 的子类")
            self._models[name] = cls
            return cls
        return decorator
    
    def get(self, name):
        """获取模型类"""
        if name not in self._models:
            raise KeyError(f"模型 '{name}' 未注册，可用: {list(self._models.keys())}")
        return self._models[name]
    
    def list_models(self):
        """列出所有模型"""
        return list(self._models.keys())
    
    def load_plugins(self, plugin_dir):
        """动态加载插件"""
        if not os.path.exists(plugin_dir):
            return
        
        # 获取包名
        package_name = os.path.basename(os.path.dirname(plugin_dir))
        plugin_package = f"{package_name}.plugins"
        
        for file in os.listdir(plugin_dir):
            if file.endswith(".py") and not file.startswith("__"):
                module_name = file[:-3]
                try:
                    importlib.import_module(f"{plugin_package}.{module_name}")
                    print(f"加载插件: {module_name}")
                except Exception as e:
                    print(f"加载 {module_name} 失败: {e}")

# 全局注册表实例
registry = ModelRegistry()
register = registry.register
get_model = registry.get
list_models = registry.list_models
